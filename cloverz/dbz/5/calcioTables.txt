CREATE TABLE campionato 
  IDCAMPIONATO   ,
  NOME  ,
  PRIMARY KEY IDCAMPIONATO


CREATE TABLE citta 
  IDCITTA   ,
  NOME  ,
  IDNAZIONE  ,
  PRIMARY KEY IDCITTA,
  CONSTRAINT citta_ibfk_1 FOREIGN KEY IDNAZIONE REFERENCES nazione IDNAZIONE


CREATE TABLE classifica 
  IDCLASSIFICA   ,
  IDCAMPIONATO  ,
  PRIMARY KEY IDCLASSIFICA,
  CONSTRAINT classifica_ibfk_1 FOREIGN KEY IDCAMPIONATO REFERENCES campionato IDCAMPIONATO


CREATE TABLE giocatore 
  IDGIOCATORE   ,
  NOME  ,
  COGNOME  ,
  VALORE  ,
  GOLFATTI  ,
  IDSQUADRA  ,
  IDNAZIONE  ,
  PRIMARY KEY IDGIOCATORE,
  CONSTRAINT giocatore_ibfk_1 FOREIGN KEY IDSQUADRA REFERENCES squadra IDSQUADRA,
  CONSTRAINT giocatore_ibfk_2 FOREIGN KEY IDNAZIONE REFERENCES nazione IDNAZIONE


CREATE TABLE giornata 
  IDGIORNATA   ,
  IDCAMPIONATO  ,
  PRIMARY KEY IDGIORNATA,
  CONSTRAINT giornata_ibfk_1 FOREIGN KEY IDCAMPIONATO REFERENCES campionato IDCAMPIONATO


CREATE TABLE nazione 
  IDNAZIONE   ,
  NOME  ,
  PRIMARY KEY IDNAZIONE


CREATE TABLE partita 
  IDPARTITA   ,
  IDCASA  ,
  IDOSPITE  ,
  GOLCASA  ,
  GOLOSPITE  ,
  ESITO  ,
  IDGIORNATA  ,
  PRIMARY KEY IDPARTITA,
  CONSTRAINT partita_ibfk_1 FOREIGN KEY IDCASA REFERENCES squadra IDSQUADRA,
  CONSTRAINT partita_ibfk_2 FOREIGN KEY IDOSPITE REFERENCES squadra IDSQUADRA,
  CONSTRAINT partita_ibfk_3 FOREIGN KEY IDGIORNATA REFERENCES giornata IDGIORNATA


CREATE TABLE squadra 
  IDSQUADRA   ,
  NOME  ,
  BUDGET  ,
  GOLFATTI  ,
  GOLSUBITI  ,
  VINTE  ,
  PERSE  ,
  PAREGGIATE  ,
  PUNTI  ,
  IDSTADIO  ,
  IDCAMPIONATO  ,
  PRIMARY KEY IDSQUADRA,
  CONSTRAINT squadra_ibfk_1 FOREIGN KEY IDSTADIO REFERENCES stadio IDSTADIO,
  CONSTRAINT squadra_ibfk_2 FOREIGN KEY IDCAMPIONATO REFERENCES campionato IDCAMPIONATO


CREATE TABLE stadio 
  IDSTADIO   ,
  NOME  ,
  CAPIENZA  ,
  COD_PROVINCIA  ,
  IDCITTA  ,
  PRIMARY KEY IDSTADIO,
  CONSTRAINT stadio_ibfk_1 FOREIGN KEY IDCITTA REFERENCES citta IDCITTA
